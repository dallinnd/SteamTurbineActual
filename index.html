packages = ["pyXSteam"] from pyXSteam.XSteam import XSteam from js import document # Initialize Table in MKS (Bar, degC, kJ/kg) steamTable = XSteam(XSteam.UNIT_SYSTEM_MKS) def calculate_turbine(*args): try: # 1. Get Inputs P_in = float(document.getElementById("P_in").value) T_in = float(document.getElementById("T_in").value) P_out = float(document.getElementById("P_out").value) m_dot = float(document.getElementById("m_dot").value) eta = float(document.getElementById("efficiency").value) # 2. Inlet State (State 1) # h_pt = Enthalpy as function of Pressure (bar) and Temp (C) h1 = steamTable.h_pt(P_in, T_in) s1 = steamTable.s_pt(P_in, T_in) # 3. Ideal Expansion (State 2s) # Isentropic: s2s = s1 # h_ps = Enthalpy as function of Pressure and Entropy h2s = steamTable.h_ps(P_out, s1) # Ideal Work w_ideal = h1 - h2s # kJ/kg # 4. Actual Expansion (State 2a) w_actual = w_ideal * eta h2a = h1 - w_actual # Define State 2a by Outlet Pressure and Actual Enthalpy t2a = steamTable.t_ph(P_out, h2a) s2a = steamTable.s_ph(P_out, h2a) x2a = steamTable.x_ph(P_out, h2a) # Format Quality if x2a == 0.0 and t2a < steamTable.tsat_p(P_out): # Check for compressed liquid (pyXSteam might return 0 for x) x_str = "Liquid" elif x2a >= 1.0 or x2a < 0: # Check for superheated (pyXSteam might return -1 or 1) x_str = "Superheated" else: x_str = f"{x2a:.3f}" delta_s = s2a - s1 power_ideal = w_ideal * m_dot power_actual = w_actual * m_dot # 6. Update UI document.getElementById("res_power_ideal").innerText = f"{power_ideal:.2f} kW" document.getElementById("res_power_actual").innerText = f"{power_actual:.2f} kW" document.getElementById("res_h2a").innerText = f"{h2a:.2f} kJ/kg" document.getElementById("res_t2a").innerText = f"{t2a:.2f} °C" document.getElementById("res_x2a").innerText = x_str document.getElementById("res_ds").innerText = f"{delta_s:.4f} kJ/kgK" document.getElementById("results-area").style.display = "block" except Exception as e: print(f"Error: {e}") document.getElementById("res_power_actual").innerText = "Error" # Hide loading message when ready document.getElementById("loading").style.display = "none"
